# Quick Start

Out of the box _@sagold/react-json-editor_ comes with a **useEditor** hook to manage data and a JSON Schema for each nested value.
In order to render any sub schema as input form, we add a set of **widgets** from _@sagold/rje-mantine-widgets_.

<Tabs>
  
  <div tabTitle="code">
    ```ts
    import { useEditor, Widget } from '@sagold/react-json-editor';
    import widgets from '@sagold/rje-mantine-widgets';
    import { MantineProvider } from '@mantine/core';

    type MyData = {
      name?: string;
      pass?: string;
    };

    function Login() {
      const editor = useEditor<MyData>({
        widgets,
        schema: {
          title: 'Login',
          type: 'object',
          required: ['name', 'pass'],
          properties: {
            name: { type: 'string', options: { icon: 'person', swapIconPosition: true } },
            pass: { type: 'string', 'x-widget': 'password' }
          }
        },
        data: { name: 'user123' },
        onChange: console.log
      });

      return (
          <MantineProvider>
              <Widget editor={editor} node={editor?.getNode()} />
          </MantineProvider>
      );
    }
    ```

  </div>
  <Canvas tabTitle="preview" className="inline" of={Introduction.Login} />
